  <p class="winner" id="winnerMsg"></p>

  <script>
    // Χώρες και σκορ
    const countries = {
      "🇺🇸 USA": 0,
      "🇹🇷 Turkey": 0,
      "🇬🇷 Greece": 0,
      "🇮🇹 Italy": 0,
      "🇬🇧 UK": 0,
      "🇦🇱 Albania": 0,
      "🌙 Arab World": 0
    };

    const scoreboard = document.getElementById("scoreboard");
    const winnerMsg = document.getElementById("winnerMsg");

    // Εμφάνιση scoreboard
    function renderScores() {
      scoreboard.innerHTML = "";
      for (const [country, score] of Object.entries(countries)) {
        const div = document.createElement("div");
        div.className = "country";
        div.textContent = `${country}: ${score} points`;
        scoreboard.appendChild(div);
      }
    }

    renderScores();

    // Προσθήκη πόντων (χειροκίνητα για demo)
    // 🌹 Rose = +1, 🍩 Donut = +20, 🌸 Perfume = νίκη γύρου
    function addPoints(country, gift) {
      if (gift === "rose") countries[country] += 1;
      else if (gift === "donut") countries[country] += 20;
      else if (gift === "perfume") {
        winnerMsg.textContent = `🏆 ${country} WINS this round with Perfume! 🌸`;
      }
      renderScores();
    }

    // 🔹 ΔΟΚΙΜΗ: πάτα στην κονσόλα για να δεις
